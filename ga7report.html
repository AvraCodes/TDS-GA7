import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np
from io import StringIO
import warnings

# Suppress warnings for cleaner output
warnings.filterwarnings('ignore')

# Set plotting style for professional visualizations
plt.style.use('seaborn-v0_8')
sns.set_palette("husl")

class EmployeeDataAnalyzer:
    """
    Comprehensive analyzer for employee department data
    Contact: 24f1002255@ds.study.iitm.ac.in
    """
    
    def __init__(self):
        # Sample dataset as provided
        self.sample_data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,R&D,Middle East,93.83,3,4.4
EMP002,Operations,North America,80.31,1,4
EMP003,Operations,Latin America,91.16,15,4.6
EMP004,HR,Europe,69.33,3,3.3
EMP005,HR,Latin America,92.83,2,3.6
EMP006,Sales,North America,85.45,5,4.2
EMP007,Sales,Europe,78.90,2,3.8
EMP008,Marketing,Asia Pacific,92.15,7,4.5
EMP009,Sales,Latin America,88.33,4,4.1
EMP010,Finance,North America,76.25,6,3.9
EMP011,Sales,Middle East,94.67,8,4.7
EMP012,R&D,Europe,89.44,12,4.3
EMP013,Operations,Asia Pacific,82.75,3,3.7
EMP014,Sales,North America,91.28,6,4.4
EMP015,HR,Latin America,73.88,1,3.2"""
        
        self.df = None
    
    def load_data(self):
        """Load employee data from CSV string"""
        try:
            self.df = pd.read_csv(StringIO(self.sample_data))
            print(f"✅ Data loaded successfully! Shape: {self.df.shape}")
            print(f"📧 Analysis by: 24f1002255@ds.study.iitm.ac.in\n")
            return True
        except Exception as e:
            print(f"❌ Error loading data: {e}")
            return False
    
    def display_sample_data(self):
        """Display first 5 rows of the dataset"""
        print("📋 SAMPLE DATA (First 5 rows):")
        print("=" * 80)
        print(self.df.head())
        print("\n")
    
    def calculate_sales_frequency(self):
        """Calculate and print frequency count for Sales department"""
        sales_count = (self.df['department'] == 'Sales').sum()
        
        print("🎯 SALES DEPARTMENT FREQUENCY ANALYSIS:")
        print("=" * 50)
        print(f"Sales Department Frequency Count: {sales_count}")
        print(f"Total Employees: {len(self.df)}")
        print(f"Sales Department Percentage: {(sales_count/len(self.df)*100):.1f}%")
        print("\n")
        
        return sales_count
    
    def get_department_statistics(self):
        """Get comprehensive department statistics"""
        dept_counts = self.df['department'].value_counts()
        dept_percentages = self.df['department'].value_counts(normalize=True) * 100
        
        print("📊 COMPLETE DEPARTMENT ANALYSIS:")
        print("=" * 50)
        
        for dept, count in dept_counts.items():
            percentage = dept_percentages[dept]
            print(f"{dept}: {count} employees ({percentage:.1f}%)")
        
        print("\n")
        return dept_counts, dept_percentages
    
    def create_department_histogram(self):
        """Create comprehensive department distribution visualization"""
        dept_counts = self.df['department'].value_counts()
        
        # Create figure with multiple subplots
        fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(16, 12))
        
        # 1. Main histogram
        colors = sns.color_palette("husl", len(dept_counts))
        bars1 = ax1.bar(dept_counts.index, dept_counts.values, color=colors, alpha=0.8, edgecolor='black')
        
        # Add value labels on bars
        for bar, count in zip(bars1, dept_counts.values):
            height = bar.get_height()
            ax1.text(bar.get_x() + bar.get_width()/2., height + 0.1,
                    f'{count}', ha='center', va='bottom', fontweight='bold', fontsize=11)
        
        ax1.set_title('Department Distribution - Employee Count', fontsize=14, fontweight='bold')
        ax1.set_xlabel('Department', fontsize=12)
        ax1.set_ylabel('Number of Employees', fontsize=12)
        ax1.tick_params(axis='x', rotation=45)
        ax1.grid(True, alpha=0.3)
        
        # Highlight Sales department
        sales_index = list(dept_counts.index).index('Sales') if 'Sales' in dept_counts.index else None
        if sales_index is not None:
            bars1[sales_index].set_color('red')
            bars1[sales_index].set_alpha(1.0)
        
        # 2. Pie chart
        wedges, texts, autotexts = ax2.pie(dept_counts.values, labels=dept_counts.index, autopct='%1.1f%%',
                                          colors=colors, startangle=90)
        ax2.set_title('Department Distribution - Percentage', fontsize=14, fontweight='bold')
        
        # 3. Horizontal bar chart
        bars3 = ax3.barh(dept_counts.index, dept_counts.values, color=colors, alpha=0.8, edgecolor='black')
        
        for bar, count in zip(bars3, dept_counts.values):
            width = bar.get_width()
            ax3.text(width + 0.1, bar.get_y() + bar.get_height()/2.,
                    f'{count}', ha='left', va='center', fontweight='bold')
        
        ax3.set_title('Department Distribution - Horizontal View', fontsize=14, fontweight='bold')
        ax3.set_xlabel('Number of Employees', fontsize=12)
        ax3.grid(True, alpha=0.3)
        
        # 4. Performance comparison by department
        dept_performance = self.df.groupby('department')['performance_score'].mean().sort_values(ascending=False)
        bars4 = ax4.bar(dept_performance.index, dept_performance.values, color=colors, alpha=0.8, edgecolor='black')
        
        for bar, score in zip(bars4, dept_performance.values):
            height = bar.get_height()
            ax4.text(bar.get_x() + bar.get_width()/2., height + 0.5,
                    f'{score:.1f}', ha='center', va='bottom', fontweight='bold')
        
        ax4.set_title('Average Performance Score by Department', fontsize=14, fontweight='bold')
        ax4.set_xlabel('Department', fontsize=12)
        ax4.set_ylabel('Average Performance Score', fontsize=12)
        ax4.tick_params(axis='x', rotation=45)
        ax4.grid(True, alpha=0.3)
        
        # Overall title and layout
        plt.suptitle('Employee Data Analysis - Department Distribution\nAnalyst: 24f1002255@ds.study.iitm.ac.in', 
                     fontsize=16, fontweight='bold', y=0.98)
        plt.tight_layout()
        plt.subplots_adjust(top=0.92)
        
        # Save the plot
        plt.savefig('department_analysis.png', dpi=300, bbox_inches='tight')
        plt.show()
    
    def generate_detailed_report(self):
        """Generate comprehensive analysis report"""
        print("📋 DETAILED ANALYSIS REPORT")
        print("=" * 60)
        print(f"Report generated by: 24f1002255@ds.study.iitm.ac.in")
        print(f"Dataset size: {len(self.df)} employees")
        print(f"Number of departments: {self.df['department'].nunique()}")
        print(f"Department list: {', '.join(sorted(self.df['department'].unique()))}")
        
        # Sales department specific analysis
        sales_data = self.df[self.df['department'] == 'Sales']
        if len(sales_data) > 0:
            print(f"\n🎯 SALES DEPARTMENT INSIGHTS:")
            print(f"Average Performance Score: {sales_data['performance_score'].mean():.2f}")
            print(f"Average Experience: {sales_data['years_experience'].mean():.1f} years")
            print(f"Average Satisfaction: {sales_data['satisfaction_rating'].mean():.1f}/5")
            print(f"Regional Distribution: {', '.join(sales_data['region'].value_counts().to_dict().keys())}")
        
        print("\n")

# Main execution function
def main():
    """Execute the complete analysis"""
    print("🏢 EMPLOYEE DEPARTMENT ANALYSIS")
    print("=" * 80)
    print("Contact: 24f1002255@ds.study.iitm.ac.in")
    print("Purpose: Calculate Sales department frequency and visualize distribution\n")
    
    # Initialize analyzer
    analyzer = EmployeeDataAnalyzer()
    
    # Load and display data
    if analyzer.load_data():
        analyzer.display_sample_data()
        
        # Calculate Sales department frequency (MAIN REQUIREMENT)
        sales_count = analyzer.calculate_sales_frequency()
        
        # Get complete department statistics
        dept_counts, dept_percentages = analyzer.get_department_statistics()
        
        # Create comprehensive visualization (MAIN REQUIREMENT)
        analyzer.create_department_histogram()
        
        # Generate detailed report
        analyzer.generate_detailed_report()
        
        print("✅ Analysis completed successfully!")
        print("📊 Histogram saved as 'department_analysis.png'")
        print("📧 Contact for questions: 24f1002255@ds.study.iitm.ac.in")
    
    else:
        print("❌ Failed to load data. Please check the dataset.")

# Execute analysis
if __name__ == "__main__":
    main()
