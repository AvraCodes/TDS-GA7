import pandas as pd
import matplotlib.pyplot as plt
import io

csv_text = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,R&D,Middle East,93.83,3,4.4
EMP002,Operations,North America,80.31,1,4
EMP003,Operations,Latin America,91.16,15,4.6
EMP004,HR,Europe,69.33,3,3.3
EMP005,HR,Latin America,92.83,2,3.6
"""

df = pd.read_csv(io.StringIO(csv_text))

sales_count = df['department'].value_counts().get('Sales', 0)
print("Frequency count for 'Sales' department:", sales_count)

codes, uniques = pd.factorize(df['department'])
plt.hist(codes, bins=range(len(uniques)+1), align='left', rwidth=0.6)
plt.xticks(range(len(uniques)), uniques)
plt.title("Histogram of Department Distribution")
plt.xlabel("Department")
plt.ylabel("Frequency")
plt.tight_layout()
plt.savefig("department_histogram.png")
