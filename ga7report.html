import pandas as pd
import matplotlib.pyplot as plt
import io, base64

csv_text = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,R&D,Middle East,93.83,3,4.4
EMP002,Operations,North America,80.31,1,4
EMP003,Operations,Latin America,91.16,15,4.6
EMP004,HR,Europe,69.33,3,3.3
EMP005,HR,Latin America,92.83,2,3.6
"""

df = pd.read_csv(io.StringIO(csv_text))

sales_count = int(df['department'].value_counts().get('Sales', 0))
print('Frequency count for Sales:', sales_count)

counts = df['department'].value_counts()
fig, ax = plt.subplots(figsize=(8,5))
ax.bar(counts.index, counts.values)
ax.set_title('Department Distribution (sample)')
ax.set_xlabel('Department')
ax.set_ylabel('Count')
fig.tight_layout()
fig.savefig('plots/department_distribution.png')
